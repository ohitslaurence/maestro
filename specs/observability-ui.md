# Observability UI Specification

**Status:** Planned\
**Version:** 1.0\
**Last Updated:** 2026-01-18

---

## 1. Overview

### Purpose

This specification defines the unified web UI for Loom's observability suite. All observability features (crashes, crons, sessions, analytics, flags) are presented in a single cohesive interface with shared components, consistent navigation, and cross-system linking.

### Goals

- **Unified experience** across all observability features
- **Cross-system linking**: Navigate from crash â†’ person â†’ flags â†’ session
- **Component library** with Storybook documentation
- **Real-time updates** via SSE integration
- **Responsive design** for desktop and tablet
- **Svelte 5** runes syntax throughout

### Non-Goals (v1)

- Mobile-first design (tablet minimum)
- Offline support
- Dark mode (deferred)
- Customizable dashboards (deferred)

```svelte
<!-- In a page/layout that needs real-time updates -->
<script lang="ts">
  import { onMount, onDestroy } from 'svelte';
  import { browser } from '$app/environment';

  let { projectId } = $props<{ projectId: string }>();

  let eventSource: EventSource | null = null;

  // Reactive state for various updates
  let recentIssues = $state<Issue[]>([]);
  let monitorStatuses = $state<Map<string, MonitorHealth>>(new Map());
  let releaseHealth = $state<ReleaseHealth | null>(null);

  onMount(() => {
    if (!browser) return;

    // Connect to unified SSE stream
    eventSource = new EventSource(`/api/projects/${projectId}/stream`);

    // Crash events
    eventSource.addEventListener('issue.new', (e) => {
      const issue = JSON.parse(e.data);
      recentIssues = [issue, ...recentIssues.slice(0, 9)];
    });

    eventSource.addEventListener('issue.regressed', (e) => {
      const data = JSON.parse(e.data);
      // Show regression notification
      showNotification({
        type: 'warning',
        title: 'Issue Regressed',
        message: `${data.short_id}: ${data.title}`,
        link: `/crashes/${data.issue_id}`,
      });
    });

    // Cron events
    eventSource.addEventListener('monitor.missed', (e) => {
      const data = JSON.parse(e.data);
      monitorStatuses.set(data.monitor_slug, 'missed');
      monitorStatuses = new Map(monitorStatuses);
    });

    eventSource.addEventListener('checkin.ok', (e) => {
      const data = JSON.parse(e.data);
      monitorStatuses.set(data.monitor_slug, 'healthy');
      monitorStatuses = new Map(monitorStatuses);
    });

    // Session events
    eventSource.addEventListener('release.health_changed', (e) => {
      releaseHealth = JSON.parse(e.data);
    });

    // Error handling
    eventSource.onerror = () => {
      // Reconnect logic handled by EventSource
      console.warn('SSE connection error, reconnecting...');
    };
  });

  onDestroy(() => {
    eventSource?.close();
  });
</script>
```

```svelte
<script lang="ts" context="module">
  import { writable } from 'svelte/store';

  export interface Notification {
    id: string;
    type: 'info' | 'success' | 'warning' | 'error';
    title: string;
    message: string;
    link?: string;
    duration?: number;
  }

  export const notifications = writable<Notification[]>([]);

  export function showNotification(notification: Omit<Notification, 'id'>) {
    const id = crypto.randomUUID();
    const duration = notification.duration ?? 5000;

    notifications.update(n => [...n, { ...notification, id }]);

    if (duration > 0) {
      setTimeout(() => {
        notifications.update(n => n.filter(item => item.id !== id));
      }, duration);
    }

    return id;
  }

  export function dismissNotification(id: string) {
    notifications.update(n => n.filter(item => item.id !== id));
  }
```
